# Guía de Integración — Código del ZIP en nuestra arquitectura
> Stack: React + Vite + TypeScript (frontend) · Django + DRF (backend) · Stripe (pagos)  
> Sin multitenancy — modelo B2C con suscripciones mensuales tipo Spotify  
> Motor de IA: ACE-Step + Qwen LLM corriendo en Modal (del ZIP, sin cambios)

---

## Decisión arquitectónica documentada

**Por qué no usamos multitenancy:**
Este sistema es B2C (Business to Consumer). Los usuarios se registran individualmente y pagan una suscripción mensual directamente. No existe el concepto de "organización" o "empresa cliente". El modelo es idéntico a Spotify o Netflix: un usuario, una suscripción, un plan. Forzar multitenancy sería añadir complejidad innecesaria que no corresponde al problema de negocio.

**Por qué Stripe en vez de Polar:**
Stripe es el estándar de la industria, el equipo ya lo conoce, tiene mejor documentación, y soporte nativo para suscripciones recurrentes con prueba gratuita, upgrades/downgrades de plan, y webhooks robustos.

### `apps/songs/services/generation_service.py`

Este es la traducción de `inngest/functions.ts` del ZIP a Python. La lógica es idéntica; solo cambia el lenguaje:

```python
# apps/songs/services/generation_service.py
"""
Traducción de la lógica de frontend/src/inngest/functions.ts del ZIP.
Orquesta: validar créditos → elegir modo → llamar Modal → guardar resultado.
"""
from django.utils import timezone
from apps.songs.models import Song, GenerationJob, Tag, SongTag
from apps.credits.services.credit_service import check_balance, deduct_credits
from apps.notifications.services import notify_user


def request_generation(user, *, title: str, description: str = None,
                       prompt: str = None, lyrics: str = None,
                       described_lyrics: str = None,
                       instrumental: bool = False,
                       guidance_scale: float = 15.0) -> GenerationJob:
    """
    Punto de entrada único para generar una canción.
    Valida créditos, crea Song + Job, encola la tarea Celery.
    """
    if not check_balance(user, required=1):
        raise InsufficientCreditsError("No tienes créditos suficientes.")

    # Determinar el modo (igual que el switch de inngest/functions.ts)
    mode = _determine_mode(description, prompt, lyrics, described_lyrics)

    # Crear canción en estado draft
    song = Song.objects.create(
        user=user,
        title=title or _default_title(description, described_lyrics),
        description=description,
        prompt=prompt,
        lyrics=lyrics,
        described_lyrics=described_lyrics,
        instrumental=instrumental,
        guidance_scale=guidance_scale,
        status='draft',
    )

    # Crear job de seguimiento
    job = GenerationJob.objects.create(
        user=user,
        song=song,
        mode=mode,
        status='queued',
    )

    # Encolar tarea Celery (en background, no bloquea la response HTTP)
    from apps.songs.tasks import process_generation_job
    process_generation_job.delay(str(job.id))

    return job


def _determine_mode(description, prompt, lyrics, described_lyrics) -> str:
    """
    Misma lógica del if/elif de inngest/functions.ts líneas 80-106.
    """
    if description:
        return 'from_description'
    elif lyrics and prompt:
        return 'with_lyrics'
    elif described_lyrics and prompt:
        return 'with_described_lyrics'
    else:
        raise ValueError("Parámetros insuficientes para determinar el modo.")


def _default_title(description: str = None, described_lyrics: str = None) -> str:
    """Misma lógica de queueSong() del ZIP líneas 39-44."""
    raw = description or described_lyrics or "Untitled"
    return raw[:1].upper() + raw[1:].strip()[:199]


class InsufficientCreditsError(Exception):
    pass
```

### `apps/songs/tasks.py`

```python
# apps/songs/tasks.py
"""
Tarea Celery equivalente a la función de Inngest del ZIP.
Referencia: frontend/src/inngest/functions.ts del ZIP.
"""
from celery import shared_task
from django.utils import timezone


@shared_task(bind=True, max_retries=2)
def process_generation_job(self, job_id: str):
    from apps.songs.models import GenerationJob
    from apps.credits.services.credit_service import deduct_credits
    from apps.notifications.services import notify_user
    from ml.modal_client import call_modal_endpoint

    try:
        job = GenerationJob.objects.select_related('user', 'song').get(id=job_id)
        song = job.song
        user = job.user

        # Paso 1: Marcar como processing
        song.status = 'processing'
        song.save(update_fields=['status'])
        job.status = 'processing'
        job.started_at = timezone.now()
        job.save(update_fields=['status', 'started_at'])

        # Paso 2: Construir endpoint y body según el modo (del ZIP inngest/functions.ts)
        endpoint_url, body = _build_modal_request(song, job.mode)
        job.modal_endpoint_used = endpoint_url
        job.save(update_fields=['modal_endpoint_used'])

        # Paso 3: Llamar a Modal (el main.py del ZIP desplegado)
        response = call_modal_endpoint(endpoint_url, body)

        # Paso 4: Guardar resultado
        song.audio_s3_key = response['s3_key']
        song.thumbnail_s3_key = response['cover_image_s3_key']
        song.status = 'ready'
        song.save(update_fields=['audio_s3_key', 'thumbnail_s3_key', 'status', 'updated_at'])

        # Paso 5: Guardar categorías/tags del LLM (del ZIP líneas 160-174)
        _save_categories(song, response.get('categories', []))

        # Paso 6: Descontar crédito SOLO si fue exitoso (como en el ZIP)
        deduct_credits(user, amount=1, reference_id=str(job.id), reference_type='generation_job')
        job.credits_used = 1
        job.status = 'completed'
        job.completed_at = timezone.now()
        job.save(update_fields=['credits_used', 'status', 'completed_at'])

        # Paso 7: Notificar al usuario
        notify_user(user, type='song_ready', reference_id=str(song.id))

    except Exception as exc:
        # Marcar como fallido (igual que onFailure de Inngest del ZIP)
        try:
            job = GenerationJob.objects.get(id=job_id)
            job.status = 'failed'
            job.error_message = str(exc)
            job.completed_at = timezone.now()
            job.save(update_fields=['status', 'error_message', 'completed_at'])
            job.song.status = 'failed'
            job.song.save(update_fields=['status'])
        except Exception:
            pass
        raise self.retry(exc=exc, countdown=60)


def _build_modal_request(song, mode: str):
    """Traduce los modos de inngest/functions.ts líneas 79-106."""
    from django.conf import settings

    common = {
        'guidance_scale': song.guidance_scale,
        'infer_step': song.infer_step,
        'audio_duration': song.audio_duration,
        'seed': song.seed,
        'instrumental': song.instrumental,
    }

    if mode == 'from_description':
        return settings.MODAL_ENDPOINT_FROM_DESCRIPTION, {
            'full_described_song': song.description,
            **common,
        }
    elif mode == 'with_lyrics':
        return settings.MODAL_ENDPOINT_WITH_LYRICS, {
            'prompt': song.prompt,
            'lyrics': song.lyrics,
            **common,
        }
    elif mode == 'with_described_lyrics':
        return settings.MODAL_ENDPOINT_FROM_DESCRIBED_LYRICS, {
            'prompt': song.prompt,
            'described_lyrics': song.described_lyrics,
            **common,
        }
    raise ValueError(f"Modo desconocido: {mode}")


def _save_categories(song, categories: list):
    """Del ZIP inngest/functions.ts líneas 160-174 — guardar tags del LLM."""
    from apps.songs.models import Tag, SongTag
    for cat_name in categories:
        tag, _ = Tag.objects.get_or_create(
            name=cat_name.strip().lower(),
            defaults={'category': 'genre'}
        )
        SongTag.objects.get_or_create(song=song, tag=tag)
```

# ml/modal_client.py
"""
Cliente HTTP para el servidor Modal desplegado desde ml/modal_music_server.py.
Referencia: frontend/src/inngest/functions.ts del ZIP (step.fetch, líneas 130-138).
"""
import requests
from django.conf import settings


def call_modal_endpoint(endpoint_url: str, body: dict) -> dict:
    """
    Llama a uno de los endpoints del MusicGenServer en Modal.
    Retorna: { s3_key, cover_image_s3_key, categories }
    """
    response = requests.post(
        endpoint_url,
        json=body,
        headers={
            "Content-Type": "application/json",
            # Autenticación de Modal (del ZIP línea 134-135)
            "Modal-Key": settings.MODAL_KEY,
            "Modal-Secret": settings.MODAL_SECRET,
        },
        timeout=600,  # ACE-Step puede tardar hasta 5-10 min en la primera llamada
    )

    if not response.ok:
        raise ModalGenerationError(
            f"Modal respondió {response.status_code}: {response.text[:300]}"
        )

    data = response.json()

    # Validar que la respuesta tiene lo que necesitamos
    if 's3_key' not in data:
        raise ModalGenerationError(f"Respuesta inesperada de Modal: {data}")

    return data


def get_presigned_url(s3_key: str, expiry_seconds: int = 3600) -> str:
    """
    Genera una URL firmada de S3 para reproducción.
    Traducción de getPresignedUrl() del ZIP (actions/generation.ts líneas 100-117).
    """
    import boto3
    from botocore.exceptions import ClientError

    s3_client = boto3.client(
        "s3",
        region_name=settings.AWS_S3_REGION_NAME,
        aws_access_key_id=settings.AWS_ACCESS_KEY_ID,
        aws_secret_access_key=settings.AWS_SECRET_ACCESS_KEY,
    )

    try:
        url = s3_client.generate_presigned_url(
            'get_object',
            Params={'Bucket': settings.AWS_STORAGE_BUCKET_NAME, 'Key': s3_key},
            ExpiresIn=expiry_seconds
        )
        return url
    except ClientError as e:
        raise ModalGenerationError(f"Error generando URL de S3: {e}")


class ModalGenerationError(Exception):
    pass

"""
Integración con Stripe para suscripciones mensuales.
Reemplaza el sistema Polar del ZIP (frontend/src/lib/auth.ts líneas 14-88).
"""
import stripe
from django.conf import settings

stripe.api_key = settings.STRIPE_SECRET_KEY

# IDs de los productos en Stripe (crear desde el dashboard de Stripe)
STRIPE_PLANS = {
    'free': {
        'credits_per_month': 5,
        'price_id': None,  # Gratis, no tiene price_id
    },
    'pro': {
        'credits_per_month': 50,
        'price_id': settings.STRIPE_PRO_PRICE_ID,    # ej: price_1Abc...
    },
    'studio': {
        'credits_per_month': 200,
        'price_id': settings.STRIPE_STUDIO_PRICE_ID, # ej: price_1Def...
    },
}


def create_checkout_session(user, plan_slug: str) -> str:
    """
    Crea una sesión de pago de Stripe.
    Retorna la URL a la que redirigir al usuario.
    """
    plan = STRIPE_PLANS.get(plan_slug)
    if not plan or not plan['price_id']:
        raise ValueError(f"Plan inválido: {plan_slug}")

    session = stripe.checkout.Session.create(
        customer_email=user.email,
        client_reference_id=str(user.id),
        mode='subscription',
        line_items=[{'price': plan['price_id'], 'quantity': 1}],
        success_url=settings.STRIPE_SUCCESS_URL,
        cancel_url=settings.STRIPE_CANCEL_URL,
        metadata={'user_id': str(user.id), 'plan': plan_slug},
    )
    return session.url


def handle_webhook(payload: bytes, sig_header: str):
    """
    Procesa eventos de Stripe webhook.
    Equivalente al onOrderPaid de Polar en auth.ts del ZIP líneas 52-85.
    """
    from apps.credits.services.credit_service import grant_credits
    from apps.users.models import User

    try:
        event = stripe.Webhook.construct_event(
            payload, sig_header, settings.STRIPE_WEBHOOK_SECRET
        )
    except stripe.error.SignatureVerificationError:
        raise ValueError("Firma de webhook inválida")

    if event['type'] == 'invoice.paid':
        invoice = event['data']['object']
        user_id = invoice.get('metadata', {}).get('user_id') or \
                  invoice.get('subscription_details', {}).get('metadata', {}).get('user_id')
        plan = invoice.get('metadata', {}).get('plan', 'pro')

        if user_id:
            try:
                user = User.objects.get(id=user_id)
                credits = STRIPE_PLANS[plan]['credits_per_month']
                grant_credits(user, amount=credits, type='monthly_grant')
            except User.DoesNotExist:
                pass

    elif event['type'] == 'customer.subscription.deleted':
        # Opcional: degradar al plan free cuando cancela
        pass

    return {'received': True}

import { create } from 'zustand';
import { Song } from '../types/song.types';

interface PlayerTrack {
  id: string;
  title: string;
  audioUrl: string | null;    // URL firmada de S3 (expira en 1h)
  thumbnailUrl: string | null;
  prompt: string | null;
  createdByUserName: string;
}

interface PlayerState {
  currentTrack: PlayerTrack | null;
  isPlaying: boolean;
  queue: PlayerTrack[];
  volume: number;

  // Acciones
  play: (track: PlayerTrack) => void;
  pause: () => void;
  resume: () => void;
  setVolume: (vol: number) => void;
  addToQueue: (track: PlayerTrack) => void;
  playNext: () => void;
}

export const usePlayerStore = create<PlayerState>((set, get) => ({
  currentTrack: null,
  isPlaying: false,
  queue: [],
  volume: 0.8,

  play: (track) => set({ currentTrack: track, isPlaying: true }),
  pause: () => set({ isPlaying: false }),
  resume: () => set({ isPlaying: true }),
  setVolume: (volume) => set({ volume }),
  addToQueue: (track) => set((state) => ({ queue: [...state.queue, track] })),
  playNext: () => {
    const { queue } = get();
    if (queue.length === 0) return;
    const [next, ...rest] = queue;
    set({ currentTrack: next, isPlaying: true, queue: rest });
  },
}));

### `src/hooks/useGenerationJob.ts`

Equivalente al polling que necesitan para saber cuándo terminó la canción:

```typescript
// src/hooks/useGenerationJob.ts
import { useEffect, useState } from 'react';
import { getGenerationJobStatus } from '../api/modules/songs.api';

type JobStatus = 'queued' | 'processing' | 'ready' | 'failed' | 'no_credits';

export function useGenerationJob(jobId: string | null) {
  const [status, setStatus] = useState<JobStatus | null>(null);
  const [songId, setSongId] = useState<string | null>(null);

  useEffect(() => {
    if (!jobId) return;

    const POLL_INTERVAL = 3000; // 3 segundos
    const FINAL_STATES: JobStatus[] = ['ready', 'failed', 'no_credits'];

    const poll = async () => {
      try {
        const data = await getGenerationJobStatus(jobId);
        setStatus(data.status);
        if (data.song_id) setSongId(data.song_id);

        if (!FINAL_STATES.includes(data.status)) {
          setTimeout(poll, POLL_INTERVAL);
        }
      } catch {
        setTimeout(poll, POLL_INTERVAL * 2); // back-off en error
      }
    };

    poll();
  }, [jobId]);

  return { status, songId };
}
```
